We start with the basis \(\b_1=1, \b_2=X, \b_3=X^2\text{.}\) Then we set
\begin{equation*} \v_1=\frac{\b_1}{\norm{\b_1}}=1. \end{equation*}
Further
\begin{equation*} \hat{\v}_2=\b_2-\langle \b_2, \v_1\rangle \cdot \v_1= X- \definite X dx{0}1 \cdot 1=X-\frac{1}{2}, \end{equation*}
\begin{equation*} \norm{\hat{\v}_2}= \definite( X-\frac{1}{2})^2dx{0}1=\frac{1}{3} (X-\frac{1}{2})^3\mid_0^1=\frac{1}{12}, \end{equation*}
\begin{equation*} \v_2=\sqrt{12}(X-\frac{1}{2}). \end{equation*}
In the next step we have
\begin{equation*} \begin{aligned} \hat{\v}_3 &=& \b_3-\langle \b_3, \v_1\rangle \cdot \v_1 - \langle \b_3, \v_2\rangle \cdot \v_2\\ &=& X^2- \definite X^2 dx{0}1 \cdot 1 - \definite X^2 \sqrt{12}(X-\frac{1}{2}) dx{0}1 \cdot \sqrt{12}(X-\frac{1}{2})\\ &=&X^2-\frac{1}{3}-12 \definite (X^3-\frac{1}{2}X^2) dx{0}1 \cdot (X-\frac{1}{2})\\ &=&X^2-\frac{1}{3}- (X-\frac{1}{2})=X^2-X+\frac{1}{6}. \end{aligned} \end{equation*}
\begin{equation*} \norm{\hat{\v}_3}=180 \end{equation*}
\begin{equation*} \v_3=\sqrt{180}(X^2-X+\frac{1}{6}). \end{equation*}
The desired \(ONB\) is \(\{1, \sqrt{12}(X-\frac{1}{2}), \sqrt{180}(X^2-X+\frac{1}{6})\}\text{.}\)
